<script type="text/javascript">
	if ( document.addEventListener ) {
        document.addEventListener( "DOMContentLoaded", setForms, false );
	} else if (window.attachEvent) {
		window.attachEvent('onload', setForms);
	}

	function setForms() {
		var allForms = document.forms;
		var firstFormFirstField = null;
		for (f = 0; f < allForms.length; f++) {
			var allInputs = allForms[f].elements;
			var textFields = [];
			for (i = 0; i < allInputs.length; i++) {
				if (allInputs[i].type === 'text' && allInputs[i].name !== 'keywords') {
					textFields.push(allInputs[i]);
				}
			}

			for (i = 0; i < textFields.length; i++) {
				// Bring focus to first form's first text field, excluding search/newsletter
				if (firstFormFirstField === null &&
					textFields[i].name &&
					textFields[i].name != 'keywords' &&
					textFields[i].name != 'ea' &&
					textFields[i].style.display != 'none') {
					firstFormFirstField = textFields[i];
				}
				textFields[i].onblur = function() {
					trimmedValue = this.value.trim();
					this.value = trimmedValue;
				}
			}
		}
		// Only if there's a form
		if (firstFormFirstField !== null) {
			firstFormFirstField.focus();
		}
	}
</script>
